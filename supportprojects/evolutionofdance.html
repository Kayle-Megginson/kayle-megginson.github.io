Evolution of Dance¶

Instructions¶

Welcome to Evolution of Dance! This program simulates a dance-off where dancers perform various dance moves randomly.

How to Play:¶

  1. Run the Program:

    • Execute the program using Python: python program.py

  2. Enter Number of Rounds:

    • Input the number of rounds you want the dance-off to last when prompted.

  3. Watch the Dance-off:

    • Enjoy watching the dancers perform different dance moves such as "The Calypso", "The Electric Slide", and more!
    • Each round, a new dancer joins the dance line and performs a random dance move.
    • Each round depending on the dance that occurs a certain amount of points will be added to your score. Depending on your score, you get a special message. Try to get them all!

  4. Play Again:

    • After the dance-off ends, you can choose to play again by entering any key except for 0. Entering '0' will quit the game.

Limitations:¶

  • Additional Customization:

    • There are no options to customize the dances or add more dancers.

  • User Interaction:

    • The game does not have interactive controls during the dance-off. It runs automatically based on the number of rounds specified at the beginning.

  • Randomness:

    • The selection of dance moves and dancers is random, which may lead to unexpected outcomes or repetitive patterns in the dance-off.

from datastructures.linked_list import LinkedList
import random


# Define the dancers and dances with emojis
dancers = ["😈", "🧌", "🌚", "💀", "🤖", "🤡", "🤠"]
dances = {
    1: "The Calypso",
    2: "The Electric Slide",
    3: "Macarena",
    4: "The Mash Potato",
    5: "The Hustle",
    6: "The Tootsie Roll"
}


# Define the scores for each dance move
dance_scores = {
    "The Calypso": 10,
    "The Electric Slide": 15,
    "Macarena": 20,
    "The Mash Potato": 10,
    "The Hustle": 15,
    "The Tootsie Roll": 20
}


# Function to generate a random dancer
def generate_random_dancer():
    return random.choice(dancers)


# Function to perform dance actions
def perform_dance_action(dance_line, dancer):
    dance = random.randint(1, 6)
    dance_move = dances[dance]
    score = dance_scores.get(dance_move, 0)  # Get the score for the dance move
    print(f"{dancer} joins and starts the \"{dance_move}\"! (Score: {score})")
    if dance == 1:
        dance_line.prepend(dancer)
    elif dance == 2:
        dance_line.append(dancer)
    elif dance == 3:
        position = random.randint(1, len(dance_line) + 1)
        current_node = dance_line.head
        for _ in range(position - 2):
            if current_node:
                current_node = current_node.next
            else:
                break
        if current_node:
            dance_line.insert_after(current_node.item, dancer)
    elif dance == 4:
        if dancer in dance_line:  # Check if the dancer exists in the list
            dance_line.extract(dancer)
            dance_line.append(dancer)
        else:
            print(f"{dancer} is not in the dance line.")
    elif dance == 5:
        if dancer in dance_line:  # Check if the dancer exists in the list
            dance_line.append(dancer)
            middle_position = len(dance_line) // 2 + 1
            current_node = dance_line.head
            for _ in range(middle_position - 2):
                if current_node:
                    current_node = current_node.next
                else:
                    break
            if current_node:
                dance_line.insert_after(current_node.item, dancer)
        else:
            print(f"{dancer} is not in the dance line.")
    elif dance == 6:
        dance_line.prepend(dancer)
        for dancer_type in dancers:
            dance_line.append(dancer_type)


# Main method
def main():
    rounds = int(input("How many rounds? "))
    dance_line = LinkedList()


    # Initialize the dance line with random dancers
    for _ in range(8):
        dance_line.append(generate_random_dancer())


    print("Your Dancers are Ready!")
    print(dance_line)
    print("The Dance Off Begins!")


    total_score = 0  # Initialize total score


    for round_num in range(1, rounds + 1):
        print(f"Round {round_num}!")
        new_dancer = generate_random_dancer()
        perform_dance_action(dance_line, new_dancer)
        print(dance_line)

        if round_num % 5 == 0:
            dance_line.extract(dance_line.front)
            dance_line.extract(dance_line.back)


    print("The Dance Off is Over!")

    # Calculate total score with random modifier
    for dance_move in dances.values():
        score = dance_scores.get(dance_move, 0)
        score += random.randint(-5, 5)  # Add a random modifier
        total_score += max(score, 0)  # Ensure score is non-negative


    # Determine performance level based on total score
    if total_score < 75:
        print("The dancers didn't quite have the sauce tonight :(")
    elif total_score <= 90:
        print("The dancers did alright... I know they can do better!")
    elif total_score < 100:
        print("The dancers did good! I'd give them an 8/10.")
    else:
        print("Wow! Absolutely incredible! That dance-off was one for the books!")


    print(f"Total Score: {total_score}")


    play_again = input("Play again? Enter any key except 0. Otherwise, enter 0 to quit ")
    if play_again != "0":
        main()


if __name__ == "__main__":
    main()
